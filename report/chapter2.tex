\begin{verbatim}
sudo sysctl -a | grep cache
\end{verbatim}

\item Model : Intel Core i7
\item Cycle time : 2.3 GHz
\item L1 i-Cache : 32 KB
\item L1 d-Cache : 32KB
\item L2 Cache (per Core) : 256 KB
\item L3 Cache : 6 MB
\end{itemize}
